# 1. reference
https://cn.vitejs.dev/guide/

# 2. create a directory
mkdir vue3-scaffold  
cd ..

# 3. init project from vite(npm 7)
npm init vite@latest vue3-scaffold -- --template vue

# 4. build and run
cd vue3-scaffold
npm install
npm run dev

or:  
yarn  // as npm install    
yarn dev // as npm dev run

# 5. check and access
curl http://localhost:3000/
